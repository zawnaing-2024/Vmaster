STEP 2: DEPLOY TO UBUNTU 22.04 PRODUCTION (40 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SSH to your Ubuntu server and run:

# Install Docker
curl -fsSL https://get.docker.com | sh
apt install -y docker-compose

# Clone from GitHub
git clone https://github.com/zawnaing-2024/Vmaster.git /var/www/vmaster
cd /var/www/vmaster

# Start containers
docker-compose -f docker-compose.prod.yml up -d
sleep 30

# Import databases
docker exec -i vmaster_db mysql -uroot -prootpassword vpn_cms_portal < database/schema.sql
docker exec -i vmaster_radius_db mysql -uradius -pradiuspass radius < radius/schema.sql

# Create admin user
docker exec vmaster_web php /var/www/html/admin/reset-admin.php

# Install Nginx + SSL
apt install -y nginx certbot python3-certbot-nginx
# (See DEPLOY_UBUNTU.md for Nginx configuration)

Access: https://your-domain.com/admin/login.php
Login: admin / admin123
